void sortNums(int *a, int cnt)
{
    int i, j, tmp;
    for(i=1; i<cnt; i++){
        tmp=a[i];
        j=i-1;
        while(tmp<a[j]&&j>=0){
            a[j+1]=a[j];
            j=j-1;
        }
        a[j+1]=tmp;
    }
        
}
